type CustomerOrderStats {
  customerid: Int!
  total_price: Float!
  num_orders: Int!
}

type Query {
  CustomerOrderStats(customerid: Int, total_price: Float, num_orders: Int): [CustomerOrderStats!]
  orders(id: Int, customerid: Int, time: String): [orders!]
}

type entries {
  productid: Int!
  quantity: Int!
  unit_price: Float!
  parent: orders!
  discount: Float!
  price: Float!
}

type orders {
  id: Int!
  customerid: Int!
  time: String!
  entries(productid: Int, quantity: Int, unit_price: Float, discount: Float, price: Float): [entries!]
  totals: totals
}

type totals {
  price: Float!
  number: Int!
  savings: Float!
  parent: orders!
}
