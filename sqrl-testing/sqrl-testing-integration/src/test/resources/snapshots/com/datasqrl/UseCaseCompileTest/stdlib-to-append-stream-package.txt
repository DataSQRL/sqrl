CREATE VIEW `_InputValues`
AS
SELECT `k`, `v`
FROM (VALUES ROW('a', 1),
   ROW('b', 2),
   ROW('a', 1),
   ROW('c', 4),
   ROW('b', 3),
   ROW('a', 4),
   ROW('c', 1),
   ROW('b', 5),
   ROW('d', 2),
   ROW('e', 1)) AS `t` (`k`, `v`);
CREATE VIEW `InputAgg`
AS
SELECT `k`, SUM(`v`) AS `v_sum`
FROM `_InputValues`
GROUP BY `k`;
CREATE VIEW `AppendStream`
AS
SELECT `k`, `v_sum`
FROM TABLE(`to_append_stream`(TABLE `InputAgg`));
