CREATE VIEW `Thing`
AS
SELECT *
FROM (VALUES ROW('a', 'label-a'),
   ROW('b', 'label-b'),
   ROW('c', 'label-c')) AS `t` (`id`, `label`);
CREATE VIEW `Detail`
AS
SELECT *
FROM (VALUES ROW('a', 'detail-a'),
   ROW('b', 'detail-b'),
   ROW('c', 'detail-c')) AS `t` (`id`, `info`);
CREATE VIEW `Tombstone`
AS
SELECT *
FROM (VALUES ROW('c')) AS `t` (`id`);
CREATE VIEW `cleanQuery`
AS
SELECT *
FROM `Thing`
WHERE `label` IS NOT NULL;
CREATE VIEW `subqueryQuery`
AS
SELECT *
FROM `Thing`
WHERE `id` NOT IN (SELECT `id`
   FROM `Tombstone`);
CREATE VIEW `joinQuery`
AS
SELECT `th`.*
FROM `Thing` AS `th`
 LEFT JOIN `Tombstone` AS `tomb` ON `th`.`id` = `tomb`.`id`
WHERE `tomb`.`id` IS NULL;
