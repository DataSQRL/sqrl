>>>flink-sql-no-functions.sql
CREATE TEMPORARY TABLE `Events__schema` (
  `event_id` BIGINT NOT NULL,
  `user_id` VARCHAR(2147483647) CHARACTER SET `UTF-16LE` NOT NULL,
  `amount` BIGINT NOT NULL,
  `event_time` TIMESTAMP(3) WITH LOCAL TIME ZONE NOT NULL
)
WITH (
  'connector' = 'filesystem',
  'format' = 'flexible-json',
  'path' = '${DATA_PATH}/events.jsonl'
);
CREATE TABLE `Events` (
  WATERMARK FOR `event_time` AS `event_time` - INTERVAL '0.001' SECOND
)
WITH (
  'source.monitor-interval' = '1m'
)
LIKE `Events__schema`;
CREATE VIEW `WindowedEventsByUser`
AS
SELECT `window_start`, `window_end`, `user_id`, COUNT(*) AS `event_count`, SUM(`amount`) AS `total_amount`
FROM TABLE(TUMBLE(TABLE `Events`, DESCRIPTOR(`event_time`), INTERVAL '5' SECOND))
GROUP BY `window_start`, `window_end`, `user_id`;
CREATE TABLE `by_user_1` (
  `window_start` TIMESTAMP(3) NOT NULL,
  `window_end` TIMESTAMP(3) NOT NULL,
  `user_id` VARCHAR(2147483647) CHARACTER SET `UTF-16LE` NOT NULL,
  `event_count` BIGINT NOT NULL,
  `total_amount` BIGINT NOT NULL,
  PRIMARY KEY (`window_start`, `window_end`, `user_id`) NOT ENFORCED
)
WITH (
  'connector' = 'print',
  'print-identifier' = 'by_user'
);
EXECUTE STATEMENT SET BEGIN
INSERT INTO `default_catalog`.`default_database`.`by_user_1`
SELECT *
 FROM `default_catalog`.`default_database`.`WindowedEventsByUser`
;
END
