USE `my_database`;

  `key_col` STRING,
  `int_col` INTEGER

  'format-version' = '2'
);
USE CATALOG `default_catalog`;
CREATE DATABASE IF NOT EXISTS `default_database`;
USE `default_database`;
CREATE VIEW `Output`
AS
SELECT *
FROM `iceberg`.`my_database`.`my_table`;
