type Package {
  name: String!
  latest(limit: Int = 10, offset: Int = 0): [Submission!]
  versions(limit: Int = 10, offset: Int = 0): [Submission!]
}

type Query {
  Package(name: String, limit: Int = 10, offset: Int = 0): [Package!]
  TopicPackages(pkgName: String, topicName: String, numSubmissions: Float, lastSubmission: String, limit: Int = 10, offset: Int = 0): [TopicPackages!]
  Submission(name: String, version: String, variant: String, latest: Boolean, type: String, license: String, repository: String, homepage: String, limit: Int = 10, offset: Int = 0): [Submission!]
  TopicSearch(topicName: String, numPackages: Float, limit: Int = 10, offset: Int = 0): [TopicSearch!]
}

type Submission {
  name: String!
  version: String!
  variant: String
  latest: Boolean!
  type: String
  license: String
  repository: String
  homepage: String
  documentation: String
  readme: String
  description: String
  uniqueId: String!
  file: String!
  hash: String!
  authorid: String!
  submissionTime: String!
  variant0: String!
  repoURL: String!
  keywords(name: String, limit: Int = 10, offset: Int = 0): [keywords!]
}

type TopicPackages {
  pkgName: String!
  topicName: String!
  numSubmissions: Float!
  lastSubmission: String!
  latest(limit: Int = 10, offset: Int = 0): [Submission!]
}

type TopicSearch {
  topicName: String!
  numPackages: Float!
}

type keywords {
  name: String!
  parent: Submission!
}
