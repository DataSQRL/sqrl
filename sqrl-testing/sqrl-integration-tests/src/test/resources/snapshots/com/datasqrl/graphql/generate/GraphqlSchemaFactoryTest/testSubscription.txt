scalar DateTime

type Mutation {
  mytable(event: mytableInput!): mytableResult!
}

  event_time: DateTime!
}

  event_time: DateTime!
}

}

  MySubscription: MySubscriptionResult!
}

  mycol: Int
  event_time: DateTime!
}

}
