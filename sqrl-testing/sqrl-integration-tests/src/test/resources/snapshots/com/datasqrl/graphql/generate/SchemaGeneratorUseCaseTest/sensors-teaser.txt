type Machine {
  machineid: Float!
  maxTemp: Float!
  avgTemp: Float!
}

type MinReadings {
  sensorid: Float!
  timeMin: String!
  temp: Float!
}

type Query {
  Sensors(id: Float, machineid: Float, placed: String): [Sensors!]
  MinReadings(sensorid: Float, timeMin: String, temp: Float): [MinReadings!]
  machinegroup(groupId: Float, groupName: String, created: String): [machinegroup!]
  Machine(machineid: Float, maxTemp: Float, avgTemp: Float): [Machine!]
  sensorreading(sensorid: Float, time: String, temperature: Float, humidity: Float): [sensorreading!]
}

type Sensors {
  id: Float!
  machineid: Float!
  placed: String!
  readings(sensorid: Float, timeMin: String, temp: Float): [MinReadings!]
}

type machinegroup {
  groupId: Float!
  groupName: String!
  created: String!
  machines(machineId: Float): [machines!]
}

type machines {
  machineId: Float!
  parent: machinegroup!
}

type sensorreading {
  sensorid: Float!
  time: String!
  temperature: Float!
  humidity: Float!
}
