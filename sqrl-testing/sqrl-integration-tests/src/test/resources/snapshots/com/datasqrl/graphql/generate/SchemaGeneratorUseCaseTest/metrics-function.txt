type Query {
  sensors(id: Float, machineid: Float, placed: String): [sensors!]
  machinegroup(groupId: Float, groupName: String, created: String): [machinegroup!]
  SecReadingByTemp(sensorid: Float, timeSec: String, temp: Float): [SecReadingByTemp!]
  SensorMaxTemp(sensorid: Float, maxTemp: Float): [SensorMaxTemp!]
  sensorreading(sensorid: Float, time: String, temperature: Float, humidity: Float): [sensorreading!]
  SecReading(sensorid: Float, timeSec: String, temp: Float): [SecReading!]
}

type SecReading {
  sensorid: Float!
  timeSec: String!
  temp: Float!
}

type SecReadingByTemp {
  sensorid: Float!
  timeSec: String!
  temp: Float!
}

type SensorMaxTemp {
  sensorid: Float!
  maxTemp: Float!
}

type machinegroup {
  groupId: Float!
  groupName: String!
  created: String!
  machines(machineId: Float): [machines!]
}

type machines {
  machineId: Float!
  parent: machinegroup!
}

type sensorreading {
  sensorid: Float!
  time: String!
  temperature: Float!
  humidity: Float!
}

type sensors {
  id: Float!
  machineid: Float!
  placed: String!
}
