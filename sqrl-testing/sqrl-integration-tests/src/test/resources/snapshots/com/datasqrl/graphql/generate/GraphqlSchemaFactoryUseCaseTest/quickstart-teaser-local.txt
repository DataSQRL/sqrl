"An RFC-3339 compliant DateTime Scalar"
scalar DateTime

type Orders {
  id: Float!
  customerid: Float!
  time: DateTime!
  items(limit: Int = 10, offset: Int = 0): [items!]
  totals(limit: Int = 10, offset: Int = 0): [totals!]
}

type Query {
  Orders(limit: Int = 10, offset: Int = 0): [Orders!]
  Users(id: Float, limit: Int = 10, offset: Int = 0): [Users!]
}

type Users {
  id: Float!
  purchases(limit: Int = 10, offset: Int = 0): [Orders!]
  spending(week: DateTime, limit: Int = 10, offset: Int = 0): [spending!]
}

type items {
  productid: Float!
  quantity: Float!
  unit_price: Float!
  discount: Float
  total: Float!
  parent: Orders!
}

type spending {
  week: DateTime!
  spend: Float!
  saved: Float!
  parent: Users!
}

type totals {
  price: Float!
  saving: Float!
  parent: Orders!
}
