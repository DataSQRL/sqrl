type Orders {
  id: Float!
  customerid: Float!
  time: String!
  items(productid: Float, quantity: Float, unit_price: Float, discount: Float, total: Float, limit: Int = 10, offset: Int = 0): [items!]
  totals(price: Float, saving: Float, limit: Int = 10, offset: Int = 0): [totals!]
}

type Query {
  Users(id: Float, limit: Int = 10, offset: Int = 0): [Users!]
  Orders(id: Float, customerid: Float, time: String, limit: Int = 10, offset: Int = 0): [Orders!]
}

type Users {
  id: Float!
  purchases(limit: Int = 10, offset: Int = 0): [Orders!]
  spending(week: String, spend: Float, saved: Float, limit: Int = 10, offset: Int = 0): [spending!]
}

type items {
  productid: Float!
  quantity: Float!
  unit_price: Float!
  discount: Float
  total: Float!
  parent: Orders!
}

type spending {
  week: String!
  spend: Float!
  saved: Float!
  parent: Users!
}

type totals {
  price: Float!
  saving: Float!
  parent: Orders!
}
