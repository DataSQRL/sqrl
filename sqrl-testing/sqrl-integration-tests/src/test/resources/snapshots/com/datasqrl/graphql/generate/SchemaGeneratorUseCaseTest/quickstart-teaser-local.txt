type Orders {
  id: Int!
  customerid: Int!
  time: String!
  items: [items!]
  totals: [totals!]
}

type Query {
  Orders(id: Int, customerid: Int, time: String): [Orders!]
  Users(id: Int): [Users!]
}

type Users {
  id: Int!
  purchases: [Orders!]
  spending: [spending!]
}

type items {
  productid: Int!
  quantity: Int!
  unit_price: Float!
  discount: Float
  parent: Orders!
  total: Float!
}

type spending {
  week: String!
  spend: Float!
  saved: Float!
  parent: Users!
}

type totals {
  price: Float!
  saving: Float!
  parent: Orders!
}
