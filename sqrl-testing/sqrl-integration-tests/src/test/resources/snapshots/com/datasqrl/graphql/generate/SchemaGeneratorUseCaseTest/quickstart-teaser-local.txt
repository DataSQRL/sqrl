type Orders {
  id: Float!
  customerid: Float!
  time: String!
  items(productid: Float, quantity: Float, unit_price: Float, discount: Float, total: Float): [items!]
  totals(price: Float, saving: Float): [totals!]
}

type Query {
  Users(id: Float): [Users!]
  Orders(id: Float, customerid: Float, time: String): [Orders!]
}

type Users {
  id: Float!
  purchases(id: Float, customerid: Float, time: String): [Orders!]
  spending(week: String, spend: Float, saved: Float): [spending!]
}

type items {
  productid: Float!
  quantity: Float!
  unit_price: Float!
  discount: Float
  total: Float!
  parent: Orders!
}

type spending {
  week: String!
  spend: Float!
  saved: Float!
  parent: Users!
}

type totals {
  price: Float!
  saving: Float!
  parent: Orders!
}
