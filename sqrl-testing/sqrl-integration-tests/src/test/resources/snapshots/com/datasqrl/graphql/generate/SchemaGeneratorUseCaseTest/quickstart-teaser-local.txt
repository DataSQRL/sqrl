type Orders {
  id: Int!
  customerid: Int!
  time: String!
  items(productid: Int, quantity: Int, unit_price: Float, discount: Float, total: Float): [items!]
  totals(price: Float, saving: Float, time: String): [totals!]
}

type Query {
  Users(id: Int, time: String): [Users!]
  Orders(id: Int, customerid: Int, time: String): [Orders!]
}

type Users {
  id: Int!
  time: String!
  purchases(id: Int, customerid: Int, time: String): [Orders!]
  spending(week: String, spend: Float, saved: Float): [spending!]
}

type items {
  productid: Int!
  quantity: Int!
  unit_price: Float!
  discount: Float
  total: Float!
  parent: Orders!
}

type spending {
  week: String!
  spend: Float!
  saved: Float!
  parent: Users!
}

type totals {
  price: Float!
  saving: Float!
  time: String!
  parent: Orders!
}
