type CustomerOrderStats {
  customerid: Int!
  total_price: Float!
  num_orders: String!
}

type Orders {
  id: Int!
  customerid: Int!
  time: String!
  entries(productid: Int, quantity: Int, unit_price: Float, discount: Float, discount0: Float, price: Float): [entries!]
  totals(price: Float, number: String, savings: Float): [totals!]
}

type Query {
  Orders(id: Int, customerid: Int, time: String): [Orders!]
  CustomerOrderStats(customerid: Int, total_price: Float, num_orders: String): [CustomerOrderStats!]
}

type entries {
  productid: Int!
  quantity: Int!
  unit_price: Float!
  discount: Float
  discount0: Float!
  price: Float!
  parent: Orders!
}

type totals {
  price: Float!
  number: String!
  savings: Float!
  parent: Orders!
}
