type CustomerOrderStats {
  customerid: Float!
  total_price: Float!
  num_orders: Float!
}

type Orders {
  id: Float!
  customerid: Float!
  time: String!
  entries(productid: Float, quantity: Float, unit_price: Float, discount: Float, discount0: Float, price: Float): [entries!]
  totals(price: Float, number: Float, savings: Float): [totals!]
}

type Query {
  Orders(id: Float, customerid: Float, time: String): [Orders!]
  CustomerOrderStats(customerid: Float, total_price: Float, num_orders: Float): [CustomerOrderStats!]
}

type entries {
  productid: Float!
  quantity: Float!
  unit_price: Float!
  discount: Float
  discount0: Float!
  price: Float!
  parent: Orders!
}

type totals {
  price: Float!
  number: Float!
  savings: Float!
  parent: Orders!
}
