type Click {
  url: String!
  timestamp: String!
  userid: String!
}

type Query {
  Trending(url: String, total: Float): [Trending!]
  Click(url: String, timestamp: String, userid: String): [Click!]
  Recommendation(url: String, rec: String, frequency: Float): [Recommendation!]
  VisitAfter(beforeURL: String, afterURL: String, timestamp: String): [VisitAfter!]
}

type Recommendation {
  url: String!
  rec: String!
  frequency: Float!
}

type Trending {
  url: String!
  total: Float!
}

type VisitAfter {
  beforeURL: String!
  afterURL: String!
  timestamp: String!
}
