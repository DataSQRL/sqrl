type Click {
  url: String!
  timestamp: String!
  userid: String!
}

type Query {
  Trending(url: String, total: Int, timestamp: String): [Trending!]
  Click(url: String, timestamp: String, userid: String): [Click!]
  Recommendation(url: String, rec: String, frequency: Int, timestamp: String): [Recommendation!]
  VisitAfter(beforeURL: String, afterURL: String, timestamp: String): [VisitAfter!]
}

type Recommendation {
  url: String!
  rec: String!
  frequency: Int!
  timestamp: String!
}

type Trending {
  url: String!
  total: Int!
  timestamp: String!
}

type VisitAfter {
  beforeURL: String!
  afterURL: String!
  timestamp: String!
}
