FROM Customer
ORDER BY (CASE
    WHEN name IS NULL THEN email
    ELSE name
END);

  LogicalProject(_uuid=[$0], _ingest_time=[$1], customerid=[$2], email=[$3], name=[$4], lastUpdated=[$5], EXPR$6=[CASE(IS NULL($4), $3, $4)])
    LogicalTableScan(table=[[customer$3]])
