type Query {
  orders(id: Int, customerid: Int, time: String): [orders!]
}

type entries {
  productid: Int!
  quantity: Int!
  unit_price: Float!
  discount: Float
  parent: orders!
}

type orders {
  id: Int!
  customerid: Int!
  time: String!
  entries(productid: Int, quantity: Int, unit_price: Float, discount: Float): [entries!]
}
