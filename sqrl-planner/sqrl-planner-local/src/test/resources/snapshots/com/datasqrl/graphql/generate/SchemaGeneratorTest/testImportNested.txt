type Orders {
  id: Float!
  customerid: Float!
  time: String!
  entries(productid: Float, quantity: Float, unit_price: Float, discount: Float, limit: Int = 10, offset: Int = 0): [entries!]
}

type Query {
  Orders(id: Float, customerid: Float, time: String, limit: Int = 10, offset: Int = 0): [Orders!]
}

type entries {
  productid: Float!
  quantity: Float!
  unit_price: Float!
  discount: Float
  parent: Orders!
}
