type Orders {
  customerid: Float!
  entries(discount: Float, productid: Float, quantity: Float, unit_price: Float): [entries!]
  id: Float!
  time: String!
}

type Product {
  category: String!
  description: String!
  name: String!
  productid: Float!
}

type Query {
  Orders(customerid: Float, id: Float, time: String): [Orders!]
  Product(category: String, description: String, name: String, productid: Float): [Product!]
}

type entries {
  discount: Float
  parent: Orders!
  productid: Float!
  quantity: Float!
  unit_price: Float!
}
